# Define a localização/idioma do programa. Atualmente existem localizações para 'en', 'pt-BR', 'it-IT', 'de-DE', 'es-AR' e 'fr-FR'.
locale = pt-BR

# Diretório temporário para processamento: "default" usa a pasta temporária do sistema.
# Configure em uma pasta livre de antivírus, indexação do sistema ou restauração. Usar um disco SSD é altamente recomendado.
indexTemp = default

# Habilite se indexTemp estiver em um disco SSD. Otimizações são feitas que podem melhorar a velocidade de processamento em até 2x.
# Não habilite se indexTemp NÃO estiver em SSD ou você terá problemas de desempenho.
indexTempOnSSD = false

# Habilite se a pasta de saída/caso estiver em SSD. Se habilitado, o índice é criado diretamente na pasta do caso,
# não no indexTemp, então você precisará de menos espaço livre na pasta temporária.
outputOnSSD = false

# Número de threads/workers de processamento: "default" usa o número de núcleos lógicos da CPU.
# Se você tiver problemas de uso de memória, pode diminuir ou aumentar a memória heap do Java (-Xms).
numThreads = default
