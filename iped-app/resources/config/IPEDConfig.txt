########################################################################
# Configurações de Processamento
########################################################################

# Habilita o cálculo de hashes dos arquivos.
enableHash = true

# Habilita o cálculo do hash photoDNA. Se você tem direito de usá-lo (forças policiais que combatem a exploração infantil)
# e recebeu a biblioteca, você deve colocá-la na pasta de plugins.
enablePhotoDNA = false

# Habilita a consulta de hashes no banco de dados do IPED.
enableHashDBLookup = false

# Habilita a consulta de photoDNA no banco de dados de hashes do IPED.
enablePhotoDNALookup = false

# Habilita a detecção de nudez. Para cada imagem, cria os atributos "nudityScore" (1 a 1000)
# e "nudityClass" (1 a 5) para classificação e/ou filtragem.
enableLedDie = false

# [Experimental] Habilita a detecção de nudez usando o algoritmo de deep learning Yahoo OpenNSFW.
# Pode ser necessário instalar o Python e algumas dependências, veja https://github.com/sepinf-inc/IPED/wiki/User-Manual#python-modules
# Isso é mais de 1 ordem de magnitude mais lento que o algoritmo LedDie na CPU, usar uma boa GPU é altamente recomendado.
# Ele atribui pontuações mais altas que o algoritmo LedDie para imagens/vídeos com genitais ou sexo explícito, principalmente por diferenças no conjunto de treinamento.
# Um atributo nsfw_nudity_score de 0 a 100 será criado.
enableYahooNSFWDetection = false

# [Experimental] Detecta Material de Abuso Sexual Infantil (CSAM) usando um modelo de IA TensorFlow, PyTorch ou ONNX.
#
# Pré-requisitos: Python e a biblioteca TensorFlow, PyTorch ou ONNX, miniaturas geradas para imagens e vídeos, hashes dos arquivos
#
# Esta funcionalidade atribui as seguintes propriedades aos arquivos de imagem e vídeo:
#     ai:csamDetector:label - classificação, valores possíveis são csam, porn ou other
#     ai:csamDetector:csam, ai:csamDetector:porn, ai:csamDetector:other - confiança de 0 a 100 para cada categoria
# Para arquivos de vídeo, as seguintes propriedades também são adicionadas:
#     ai:csamDetector:triggerFrame - número do quadro com maior confiança entre as miniaturas do vídeo
#     ai:csamDetector:hitPercentage - percentual do total de quadros analisados das categorias csam ou porn
#     ai:csamDetector:avgConfidence - média de todas as confianças da categorização escolhida
#
# Para alterar o modelo a ser usado e outros parâmetros, edite o arquivo conf/CSAMDetector.txt
#
enableCSAMDetector = false

# Habilita a detecção e decodificação de QRCode.
enableQRCode = false

# Ignora e exclui arquivos duplicados com o mesmo hash do processamento e do caso.
# ATENÇÃO: Esta funcionalidade é inerentemente PERIGOSA, pois pode excluir do caso arquivos duplicados com
# metadados diferentes (nome, caminho, datas) como anexos de e-mail ou arquivos em pastas relevantes. Veja #574 e #931.
ignoreDuplicates = false

# Exporta as propriedades de todos os arquivos do caso para o arquivo "Lista de Arquivos.csv"
exportFileProps = true

# Processa as assinaturas dos arquivos.
processFileSignatures = true

# Habilita o parsing do conteúdo dos arquivos. É necessário para diversas funções como
# expansão de contêineres, detecção de conteúdo criptografado, indexação de metadados, geração de pré-visualizações, etc.
enableFileParsing = true

# Expande contêineres, como arquivos compactados, caixas de e-mail, e-mails e documentos MS-Office.
# Os tipos de arquivo a serem expandidos podem ser modificados no arquivo "conf/CategoriesToExpand.txt"
expandContainers = true

# Processa imagens de disco suportadas (como dd, e01, vhd, vmdk) contidas na evidência principal recursivamente.
# Imagens divididas e snapshots não são suportados por enquanto.
processEmbeddedDisks = true

# Pesquisa textos extraídos dos itens por expressões regulares
# como CPF, e-mails, URLs, cartões de crédito, valores monetários, etc.
# Novas expressões podem ser configuradas no arquivo "conf/RegexConfig.txt"
enableRegexSearch = true

# Habilita a exportação automática de arquivos para a pasta do caso baseado em categorias ou palavras-chave.
# Se habilitado, apenas os arquivos exportados são incluídos no caso, os demais são ignorados, e o caso se torna independente da fonte de dados.
# Categorias e palavras-chave relevantes podem ser configuradas em CategoriesToExport.txt e KeywordsToExport.txt.
enableAutomaticExportFiles = false

# Habilita a detecção de mais de 70 idiomas em arquivos de documentos.
enableLanguageDetect = true

# Funcionalidade para reconhecer entidades nomeadas: pessoas, organizações ou lugares.
# Os modelos de linguagem do StanfordCoreNLP 3.8 devem ser baixados de https://stanfordnlp.github.io/CoreNLP/history.html e colocados na pasta de plugins.
# Esta funcionalidade consome muita CPU, podendo aumentar o tempo de processamento em até 4x.
# Configurações avançadas podem ser modificadas no arquivo "conf/NamedEntityRecognitionConfig.txt".
enableNamedEntityRecogniton = false

# Funcionalidade para criar grafos de vínculos de comunicações (chamadas, mensagens, e-mails...)
# Opcionalmente pode criar entidades e vínculos baseados em ocorrências próximas de expressões regulares (experimental). Pode ser habilitado em conf/GraphConfig.json
enableGraphGeneration = true

# Realiza teste de aleatoriedade em itens binários e tipos de arquivo desconhecidos.
# Isto é usado para detectar possíveis itens criptografados.
# Também torna a indexação mais rápida e reduz o tamanho do índice, especialmente ao indexar espaço não alocado.
# Pode causar perda de ocorrências cercadas por conteúdo "aleatório".
entropyTest = true

# Indexa o conteúdo dos arquivos. Se desabilitado, indexa apenas as propriedades dos arquivos.
indexFileContents = true

# Habilita a indexação em um cluster ElasticSearch/OpenSearch.
# Você deve configurar pelo menos as opções de host/porta em conf/ElasticSearchConfig.txt
enableIndexToElasticSearch = false

# Habilita a exportação de arquivos para o cluster de armazenamento de objetos MinIO.
# Você deve configurar pelo menos as opções de host/porta em conf/MinIOConfig.txt
enableMinIO = false

# Habilita OCR em imagens e arquivos PDF digitalizados. Pode aumentar muito o tempo de processamento.
# O dicionário de idioma do OCR e outras configurações avançadas podem ser encontrados em conf/OCRConfig.txt.
enableOCR = false

# Habilita a transcrição de áudio.
# A implementação padrão usa a transcrição VOSK na CPU local (mais rápida, mas com menor precisão).
# Você pode alterar o algoritmo em conf/AudioTranscriptConfig.txt:
# - Algoritmo Wav2Vec2 (mais lento e boa precisão)
# - Algoritmo Whisper (muito mais lento, mas melhor precisão)
# - Google Cloud (custo de aproximadamente US$1,00 por hora)
# - Microsoft Azure (custo de aproximadamente US$1,00 por hora)
enableAudioTranscription = false

# Habilita o data carving. "addUnallocated" deve estar habilitado para varrer o espaço não alocado.
# Por padrão, nosso módulo de carving varre muitos locais na evidência em busca de arquivos deletados ou embutidos.
# Locais a serem varridos e tipos de arquivo a serem recuperados podem ser configurados em "conf/CarverConfig.xml"
enableCarving = false

# Habilita o carving que recupera arquivos conhecidos da base LED, baseado no início (64K) do arquivo.
# É necessário habilitar "addUnallocated" e configurar "hashesDB" (com dados de hashes LED importados).
enableLedCarving = false

# Habilita o carving específico de arquivos "known.met" do e-Mule. "addUnallocated" deve estar habilitado.
enableKnownMetCarving = false

# Habilita a geração de miniaturas de imagens durante o processamento.
# Pode tornar o processamento mais lento, mas acelera significativamente a pré-visualização de imagens na galeria.
enableImageThumbs = true

# Habilita a geração de características de imagens durante o processamento, permitindo a busca
# por imagens visualmente similares. "enableImageThumbs" deve estar habilitado.
enableImageSimilarity = false

# Habilita a funcionalidade de reconhecimento facial.
# Pode ser necessário instalar o Python e algumas dependências, veja https://github.com/sepinf-inc/IPED/wiki/User-Manual#facerecognition
# Opções de configuração avançadas podem ser encontradas em conf/FaceRecognitionConfig.txt.
# Se habilitado, você pode buscar faces a partir da interface de análise, verifique o menu de opções.
enableFaceRecognition = false

# Habilita a funcionalidade de estimativa de idade.
# Pode ser necessário instalar o Python e algumas dependências, veja https://github.com/sepinf-inc/IPED/wiki/User-Manual#AgeEstimation
# Opções de configuração avançadas podem ser encontradas em conf/AgeEstimationConfig.txt.
enableAgeEstimation = false

# Habilita a busca por carteiras de hardware de criptomoedas.
enableSearchHardwareWallets = true

# Habilita a extração de quadros de imagem de arquivos de vídeo.
# As configurações de extração podem ser modificadas no arquivo "conf/VideoThumbsConfig.txt"
enableVideoThumbs = true

# [Experimental] Habilita a criação de miniaturas para PDFs e diversos formatos suportados pelo LibreOffice.
# Configurações avançadas podem ser encontradas em conf/DocThumbsConfig.txt.
enableDocThumbs = false

# Habilita a geração de relatórios HTML em extrações automáticas ou a partir de itens selecionados.
# As configurações de geração podem ser modificadas no arquivo "conf/HTMLReportConfig.txt"
enableHTMLReport = true

# Habilita o classificador de imagens remoto.
# Realiza a classificação de arquivos de imagem e vídeo.
# Configurações avançadas podem ser modificadas no arquivo "conf/RemoteImageClassifierConfig.txt"
enableRemoteImageClassifier = false
